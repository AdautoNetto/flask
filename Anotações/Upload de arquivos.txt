		abrir o arquivo init.py

			app.config['UPLOAD_FILES'] = r'static/data'
	
	alterar o MODELS para receber o nome do arquivo
		abrir o arquivo models.py
		dentro da classe Post crio mais um campo para imagem
			imagem = db.Column(db.String, nullable=True, default='default.png')
			lembrar de da o flask db migrate e upgrade

	alterar o Form para salvar o arquivo na pasta e o nome no banco de dados
		abrir o arquivo forms.py
			fazer a importação das seguintes ferramentas e bibliotecas
				from wtforms import FileField
				import os -> para salvar
				from werkzeug.utils import secure_filename -> para deixar o nome seguro
			criar o campo imagem na classe PostForm
				imagem = FileField('imagem', validators=[DataRequired()])
			configurar o campo criado dentro das funções para salvar a imagem
				para pegar a imagem e trabalhar com ela 
					imagem = self.imagem.data
				para colocar um nome seguo na imagem
					nome_seguro = secure_filename(imagem.filename)
				adicionar imagem = nome_seguro dentro do post = Post
					post = Post (
                				mensagem = self.mensagem.data,
                				user_id = user_id,
                				imagem = nome_seguro
            					)
			para pegar o caminho que eu vou salvar o arquivo, fazer esse passo a passo dentro da função save
				caminho = os.path.join(
                 			os.path.abspath(os.path.dirname(__file__)) #pegar a pasta que está no nosso projeto 
                 			app.config['UPLOAD_FILES'] #definir a pasta que configuramos para UPLOAD
                 			'post', nome_seguro #defino nome seguro pra ela e a pasta que está os POST  
            				)
            				imagem.save(caminho) #salvar no bd
